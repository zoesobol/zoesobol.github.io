const fs = require('fs')

let deptosJson = fs.readFileSync('./departamentos.json');

let deptos = JSON.parse(deptosJson)

let inmobiliaria = {
    deptos:deptos,
    departamentosDisponibles: function() {
        return this.deptos.filter(depto => depto.disponible)
    },
    listarDepartamentos: function() {
        this.deptos.forEach(depto => {
            console.log(`DEPTO ${depto.id}:\nPrecio: $${depto.precioAlquiler}`)
            if(depto.disponible){
                console.log('Está disponible.')
            } else {
                console.log('Está alquilado')
            }
            console.log('-----------')
        });
    },
    buscarPorId: function(id) {
        return this.deptos.find(depto => id === depto.id)
    },
    buscarPorPrecio: function(precio) {
        return this.deptos.filter(depto => depto.precioAlquiler <= precio)
    },
    ordenarPorPrecio: function() {
        let arrayOrdenado = this.deptos.sort((deptoA, deptoB) => 
            (deptoA.precioAlquiler > deptoB.precioAlquiler) ? 1 : -1)
        return arrayOrdenado;
    },
    precioConImpuesto: function(porcentaje) {
        let arrayConImpuesto = this.deptos.map(depto => depto.precioAlquiler + depto.precioAlquiler * porcentaje / 100)
        return arrayConImpuesto;
    },
    alquilarDepto: function(id){
        let deptoAAlquilar = this.buscarPorId(id);
        deptoAAlquilar.disponible = false;
    },
    paguenRatas: function() {
        let suma = this.deptos.reduce((total,depto) => (total + depto.precioAlquiler),0)
        return suma.toFixed(2);
    }
}

let carrera = {
    autos:autos,
    autosPorTanda:5,
    autosHabilitados: function() {
        return this.autos.filter(auto => !auto.sancionado);
    },
    listarAutos: function() {
        this.autos.forEach(auto => {
            if(auto.sancionado) {
                console.log(`Piloto: ${auto.piloto}, Patente: ${auto.patente}, peso en kg: ${auto.peso}, estado: sancionado`);
            } else {
                console.log(`Piloto: ${auto.piloto}, Patente: ${auto.patente}, peso en kg: ${auto.peso}, estado: puede correr`);
            }
        })
    },
    buscarPorPiloto: function(piloto) {
        return this.autos.find(auto => piloto === auto.piloto);
    },
    buscarPorAceleracion: function(aceleracion) {
        let autosHabilitados = this.autosHabilitados();
        return autosHabilitados.filter(auto => auto.aceleracion >= aceleracion);
    },
    ordenarPorAngulo: function() {
        let arrayOrdenado = this.autos.sort((autoA, autoB) => 
            (autoA.anguloDeGiro > autoB.anguloDeGiro) ? 1 : -1);
        return arrayOrdenado;
    },
    generarTanda: function(cilindrada,peso) {
        let autosHabilitados = this.autosHabilitados();
        let autosQueCumplenLasCondiciones = autosHabilitados.filter(auto => auto.cilindrada <= cilindrada && auto.peso <= peso);
        return autosQueCumplenLasCondiciones.slice(0,this.autosPorTanda);
    },
    calcularPodio: function(cilindrada,peso) {
        let podio = this.generarTanda(cilindrada,peso)
        podio.sort((autoA, autoB) => 
            (autoA.puntaje > autoB.puntaje) ? 1 : -1);
        podio.reverse();
        console.log(`EL GANADOR ES ${podio[0].piloto}, con un puntaje de ${podio[0].puntaje}`);
        console.log(`El segundo puesto es para ${podio[1].piloto}, con un puntaje de ${podio[1].puntaje}`);
        console.log(`Y el tercer puesto es para ${podio[2].piloto}, con un puntaje de ${podio[2].puntaje}`)
    }
}
